CREATE TABLE PASSENGER
(SSN int(10) NOT NULL,Name varchar(255) NOT NULL,ADDRESS varchar(255) NOT NULL,EMAIL ID varchar(255) NOT NULL,PHONE NUMBER int(10) NOT NULL);

CREATE TABLE TICKET
(TICKET ID int(7) NOT NULL,SEAT NUMBER int(3) NOT NULL);

CREATE TABLE PAYMENT
(PAYMENT ID int(10) NOT NULL,PAYMENT AMOUNT int NOT NULL);

CREATE TABLE CARD DETAILS
(CARD NUMBER int(12) NOT NULL,EXPIRATION DATE date NOT NULL,CVV int(4) NOT NULL);

CREATE TABLE FLIGHT
(FLIGHT NUMBER int(5) NOT NULL,FLIGHT TYPE varchar(255) NOT NULL,CAPACITY int(3) NOT NULL);

CREATE TABLE FLIGHT SCHEDULE
(ARRIVALTIME time NOT NULL,DEPARTURETIME time NOT NULL,ARRIVAL LOCATION varchar(255) NOT NULL,DEPARTURE LOCATION varchar(255) NOT NULL);

CREATE TABLE AIRPORT
(AIRPORT ID int(4) NOT NULL,AIRPORT NAME varchar(255) NOT NULL,LOCATION varchar(255) NOT NULL);

CREATE TABLE EMPLOYEE
(SSN int(10) NOT NULL,EMPLOYEE ID int(7) NOT NULL,NAME varchar(255) NOT NULL,PHONE NUMBER int(10) NOT NULL,EMAIL ID varchar(255) NOT NULL);
